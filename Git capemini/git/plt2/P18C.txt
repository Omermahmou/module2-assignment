import java.util.Scanner;
public class P18C {
   public static void main(String[] args) {
      Scanner sc=new Scanner(System.in);
      System.out.println("Enter Row Number: ");
      int row=sc.nextInt();
      for (int i = 0; i < row; i++) {
         for (int j = 1; j <= 5; j++) {
            System.out.print(j);
         }
         System.out.println();
      }
   }
}
