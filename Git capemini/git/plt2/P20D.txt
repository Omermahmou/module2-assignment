import java.util.Scanner;
public class P20D {
   public static void main(String[] args) {
      Scanner sc=new Scanner(System.in);
      System.out.println("Enter Row: ");
      int row=sc.nextInt();
      for(int i=0; i<row; i++)
      {
         for (int j=1; j<=(i * 2) -1; j++)
         {
            System.out.print("*");
         }
         System.out.println();
      }
   }
}
