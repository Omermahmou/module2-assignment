import java.util.*;
class customer{
   String name;
   double price;
   String itemCode;
   String description;
   int qty;
   double initialPrice=0;
   double total;

   void input(){
      Scanner sc=new Scanner(System.in);
      System.out.println("Enter Customer Name: ");
      name=sc.nextLine();
      System.out.println("Enter Item Code: ");
      itemCode=sc.next();
      System.out.println("Enter Description: ");
      description=sc.next();
      System.out.println("Enter Quantity: ");
      qty=sc.nextInt();
      System.out.println("Enter Price: ");
      price=sc.nextDouble();
   }
   double finalAmount(){
      initialPrice=(qty*price);
      if (initialPrice>10000){
         total=initialPrice-initialPrice*0.10;
      }
      else if (initialPrice<1000){
         total=initialPrice-initialPrice*0.25;
      }
      return total;
   }
   void display(){
      System.out.println("Customer Name: "+name);
      System.out.println("Product Code: "+itemCode);
      System.out.println("Description: "+description);
      System.out.println("Quantity: "+qty);
      System.out.println("Final Price: "+total);
   }
}
public class Main
{

   public static void main(String args[]){
   Scanner sc=new Scanner(System.in);
      System.out.println("1.Enter 'Y' for continue");
      String op=sc.next();
      System.out.println("2.Enter 'N' for exit");
      int choice=Character.getNumericValue(op.charAt(0));
      switch (choice){
         case 1:
            customer c=new customer();
            c.input();
            c.display();
            break;
         case 2:
            break;
      }
   }
}
