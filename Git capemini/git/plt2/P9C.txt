import java.util.Scanner;
public class P9C {

   public static void main(String[] args) {
      Scanner sc=new Scanner(System.in);
      System.out.println("Enter N value: ");
      int n=sc.nextInt();
      int j=1,k=2;
      for (int i=0;i<n;i++){
         if (i%2 !=0){
            j=j+3;
            System.out.println(j+" ");
         }
         else if (i%2==0){
            k=k+4;
            System.out.println(k+" ");
         }
      }
   }
}
